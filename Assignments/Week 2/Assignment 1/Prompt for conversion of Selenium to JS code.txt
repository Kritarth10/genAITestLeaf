       const { chromium } = require('playwright'); // Or firefox, webkit

async function runPlaywright() {
    const browser = await chromium.launch(); // Or firefox.launch(), webkit.launch()
    const page = await browser.newPage();
    await page.goto('https://www.example.com');
    const headingText = await page.locator('#main-heading').textContent();
    console.log('Heading text (Playwright):', headingText);
    await browser.close();
}
